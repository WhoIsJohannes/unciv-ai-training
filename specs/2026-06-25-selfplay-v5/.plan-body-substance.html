Behavioral delta — micro-batch epoch (pseudocode; math frozen, traversal chunked)

opt.zero_grad()
if micro_batch_steps and micro_batch_steps < N:          # chunked path
    loss_total = 0.0
    for (lo, hi) in chunks_of(N, micro_batch_steps):
        tl, pl, val = forward_chunk_fn(lo, hi)            # net(sliced inputs) — per-row independent
        pol_c, val_c, ent_c = _ac_loss_terms(tl, pl, val.reshape(-1),
                                  a_tech[lo:hi], a_policy[lo:hi], m_tech[lo:hi], m_policy[lo:hi],
                                  adv[lo:hi], ret[lo:hi], old_logp[lo:hi], clip_eps)   # SAME expressions
        loss_c = (pol_c + value_coef*val_c - entropy_coef*ent_c) * ((hi-lo)/N)         # size-weight
        loss_c.backward()                                 # accumulate grad, free chunk graph
        loss_total += loss_c.detach()
    loss = loss_total
else:                                                     # whole-batch path — UNCHANGED from v4
    tl, pl, val = forward_fn()
    pol, val_l, ent = _ac_loss_terms(...)                 # full batch
    loss = pol + value_coef*val_l - entropy_coef*ent
    loss.backward()
if not torch.isfinite(loss):                              # divergence guard (now opt-aware)
    net.load_state_dict(safe_state); opt.load_state_dict(safe_opt); stats["diverged"]=True; return
grad_norm = clip_grad_norm_(net.parameters(), max_norm=10.0)   # once, on accumulated grad
opt.step(); safe_state = deepcopy(net.state_dict()); safe_opt = deepcopy(opt.state_dict())

adv/ret/old_logp are computed ONCE on the full flattened batch (adv is normalized batch-wide) and only indexed per chunk. Size-weighting (hi-lo)/N makes the summed per-chunk .mean()s equal the whole-batch .mean() within fp tolerance.
